51  /**
52   * Generates SCoverage report in forked {@code scoverage} life cycle.
53   * <br/>
54   * <br/>
55   * In forked {@code scoverage} life cycle project is compiled with SCoverage instrumentation
56   * and tests are executed before report generation.
57   * <br/>
58   * 
59   * @author <a href="mailto:gslowikowski@gmail.com">Grzegorz Slowikowski</a>
60   * @since 1.0.0
61   */
62  @Mojo( name = "report", threadSafe = false )
63  @Execute( lifecycle = "scoverage", phase = LifecyclePhase.TEST )
64  public class SCoverageReportMojo
65      extends AbstractMojo
66      implements MavenReport

156     /** {@inheritDoc} */
157     @Override
158     public void generate( Sink sink, Locale locale )
159         throws MavenReportException
160     {
161         try
162         {
163             long ts = System.currentTimeMillis();
164 
165             File sourceDir = new File( project.getBuild().getSourceDirectory() );
166 
167             mkdirs( outputDirectory );
168             mkdirs( xmlOutputDirectory );
169             
170             /*does not work, because empty "index.html" is already there and cannot be deleted.
171             File reportOutputDir = getReportOutputDirectory();
172             if ( reportOutputDir.exists() )
173             {
174                 if ( reportOutputDir.isDirectory() )
175                 {
176                     try
177                     {
178                         FileUtils.deleteDirectory( reportOutputDir );
179                     }
180                     catch ( IOException e)
181                     {
182                         throw new MavenReportException( String.format( "Cannot delete \"%s\" directory", reportsDir.getAbsolutePath() ), e ); 
183                     }
184                 }
185                 else
186                 {
187                    throw new MavenReportException( String.format( "\"%s\" is not a directory", reportsDir.getAbsolutePath() ) ); 
188                 }
189             }*/
190 
191             File coverageFile = Serializer.coverageFile( dataDirectory );
192             Coverage coverage = Serializer.deserialize( coverageFile );
193 
194             File[] measurementFiles = IOUtils.findMeasurementFiles( dataDirectory );
195             scala.collection.Set<Object> measurements = IOUtils.invoked( Predef$.MODULE$
196                 .wrapRefArray( measurementFiles ) );
197             coverage.apply( measurements );
198 
199             getLog().info( "[scoverage] Generating cobertura XML report..." );
200             new CoberturaXmlWriter( project.getBasedir(), xmlOutputDirectory ).write( coverage );
201 
202             getLog().info( "[scoverage] Generating scoverage XML report..." );
203             new ScoverageXmlWriter( sourceDir, xmlOutputDirectory, false ).write( coverage );
204 
205             getLog().info( "[scoverage] Generating scoverage HTML report..." );
206             new ScoverageHtmlWriter( sourceDir, outputDirectory ).write( coverage );
207 
208             long te = System.currentTimeMillis();
209             getLog().debug( String.format( "Mojo execution time: %d ms", te - ts ) );
210         }
211         catch ( RuntimeException e )
212         {
213             if ( failOnError )
214             {
215                 throw e;
216             }
217             getLog().error( "Error while creating javadoc report: " + e.getMessage(), e );
218         }
219     }

235     /** {@inheritDoc} */
236     @Override
237     public boolean canGenerateReport()
238     {
239         if ( skip )
240         {
241             return false;
242         }
243 
244         if ( "pom".equals( project.getPackaging() ) )
245         {
246             return false;
247         }
248 //        return !"pom".equals( project.getPackaging() );
249         //return true;//TODO-don't generate for aggregator nodes 
250 
251         File coverageFile = Serializer.coverageFile( dataDirectory );
252         if ( !coverageFile.exists() || !coverageFile.isFile() )
253         {
254 //            getLog().warn( "[scoverage] No coverage data, report generation skipped." );
255             return false;
256         }
257         return true;
258     }

281     private void updateReportOutputDirectory( File reportOutputDirectory )
282     {
283         if ( reportOutputDirectory != null && destDir != null
284              && !reportOutputDirectory.getAbsolutePath().endsWith( destDir ) )
285         {
286             this.outputDirectory = new File( reportOutputDirectory, destDir );
287         }
288         else
289         {
290             this.outputDirectory = reportOutputDirectory;
291         }
292     }
293 
294     /**
295      * Generates SCoverage report.
296      * 
297      * @throws MojoExecutionException if unexpected problem occurs
298      */
299     @Override
300     public void execute()
301         throws MojoExecutionException
302     {
303         if ( !canGenerateReport() )
304         {
305             getLog().info( "Skipping SCoverage report generation" );
306             return;
307         }
308 
309         try
310         {
311             RenderingContext context = new RenderingContext( outputDirectory, getOutputName() + ".html" );
312             SiteRendererSink sink = new SiteRendererSink( context );
313             Locale locale = Locale.getDefault();
314             generate( sink, locale );
315         }
316         catch ( MavenReportException e )
317         {
318             failOnError( "An error has occurred in " + getName( Locale.ENGLISH ) + " report generation", e );
319         }
320         catch ( RuntimeException e )
321         {
322             failOnError( "An error has occurred in " + getName( Locale.ENGLISH ) + " report generation", e );
323         }
324     }

326     /**
327      * Gets the resource bundle for the specified locale.
328      *
329      * @param locale The locale of the currently generated report.
330      * @return The resource bundle for the requested locale.
331      */
332     private ResourceBundle getBundle( Locale locale )
333     {
334         return ResourceBundle.getBundle( "scoverage-report", locale, getClass().getClassLoader() );
335     }
336 
337     private void failOnError( String prefix, Exception e )
338         throws MojoExecutionException
339     {
340         if ( failOnError )
341         {
342             if ( e instanceof RuntimeException )
343             {
344                 throw (RuntimeException) e;
345             }
346             throw new MojoExecutionException( prefix + ": " + e.getMessage(), e );
347         }
348 
349         getLog().error( prefix + ": " + e.getMessage(), e );
350     }
351 
352     private void mkdirs( File directory )
353         throws MavenReportException
354     {
355         if ( !directory.exists() && !directory.mkdirs() )
356         {
357             throw new MavenReportException( String.format( "Cannot create \"%s\" directory ",
358                                                            directory.getAbsolutePath() ) );
359         }
360     }
