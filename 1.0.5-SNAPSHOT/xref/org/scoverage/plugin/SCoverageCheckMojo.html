18  package org.scoverage.plugin;
19  
20  import java.io.File;
21  
22  import org.apache.maven.plugin.AbstractMojo;
23  import org.apache.maven.plugin.MojoFailureException;
24  import org.apache.maven.plugins.annotations.Execute;
25  import org.apache.maven.plugins.annotations.LifecyclePhase;
26  import org.apache.maven.plugins.annotations.Mojo;
27  import org.apache.maven.plugins.annotations.Parameter;
28  import org.apache.maven.project.MavenProject;
29  
30  import scala.Predef$;
31  
32  import scoverage.Coverage;
33  import scoverage.IOUtils;
34  import scoverage.Serializer;
35  
36  /**
37   * Checks if minimum code coverage by unit tests reached
38   * in forked {@code scoverage} life cycle.
39   * <br/>
40   * <br/>
41   * In forked {@code scoverage} life cycle project is compiled with SCoverage instrumentation
42   * and unit tests are executed before checking.
43   * <br/>
44   * 
45   * @author <a href="mailto:gslowikowski@gmail.com">Grzegorz Slowikowski</a>
46   * @since 1.0.0
47   */
48  @Mojo( name = "check", threadSafe = false )
49  @Execute( lifecycle = "scoverage", phase = LifecyclePhase.TEST )
50  public class SCoverageCheckMojo
51      extends AbstractMojo
52  {
53  
54      /**
55       * Allows SCoverage to be skipped.
56       * <br/>
57       * 
58       * @since 1.0.0
59       */
60      @Parameter( property = "scoverage.skip", defaultValue = "false" )
61      private boolean skip;
62  
63      /**
64       * Directory where the coverage files should be written.
65       * <br/>
66       *
67       * @since 1.0.0
68       */
69      @Parameter( property = "scoverage.dataDirectory", defaultValue = "${project.build.directory}/scoverage-data", required = true, readonly = true )
70      private File dataDirectory;
71  
72      /**
73       * Required minimum coverage.
74       * <br/>
75       * <br/>
76       * See <a href="https://github.com/scoverage/sbt-scoverage#minimum-coverage">https://github.com/scoverage/sbt-scoverage#minimum-coverage</a> for additional documentation.
77       * <br/>
78       *
79       * @since 1.0.0
80       */
81      @Parameter( property = "scoverage.minimumCoverage", defaultValue = "0" )
82      private Double minimumCoverage;
83  
84      /**
85       * Fail the build if minimum coverage was not reached.
86       * <br/>
87       * <br/>
88       * See <a href="https://github.com/scoverage/sbt-scoverage#minimum-coverage">https://github.com/scoverage/sbt-scoverage#minimum-coverage</a> for additional documentation.
89       * <br/>
90       * 
91       * @since 1.0.0
92       */
93      @Parameter( property = "scoverage.failOnMinimumCoverage", defaultValue = "false" )
94      private boolean failOnMinimumCoverage;
95  
96      /**
97       * Maven project to interact with.
98       */
99      @Parameter( defaultValue = "${project}", readonly = true, required = true )
100     private MavenProject project;
101 
102     /**
103      * Checks tests coverage and optionally fails the build if minimum level not reached.
104      * 
105      * @throws MojoFailureException if coverage is below minimumCoverage and failOnMinimumCoverage option set
106      */
107     @Override
108     public void execute() throws MojoFailureException
109     {
110         if ( "pom".equals( project.getPackaging() ) )
111         {
112             getLog().info( "Skipping SCoverage execution for project with packaging type 'pom'" );
113             //for aggragetor mojo - list of submodules: List<MavenProject> modules = project.getCollectedProjects();
114             return;
115         }
116 
117         if ( skip )
118         {
119             getLog().info( "Skipping Scoverage execution" );
120             return;
121         }
122         
123         long ts = System.currentTimeMillis();
124 
125         if ( !dataDirectory.exists() || !dataDirectory.isDirectory() )
126         {
127             getLog().info( "Cannot perform check, instrumentation not performed - skipping" );
128             return;
129         }
130 
131         File coverageFile = Serializer.coverageFile( dataDirectory );
132         if ( !coverageFile.exists() )
133         {
134             getLog().info( "Scoverage data file does not exist. Skipping report generation" );
135             return;
136         }
137         if ( !coverageFile.isFile() )
138         {
139             getLog().info( "Scoverage data file is a directory, not a file. Skipping report generation" );
140             return;
141         }
142 
143         Coverage coverage = Serializer.deserialize( coverageFile );
144         File[] measurementFiles = IOUtils.findMeasurementFiles( dataDirectory );
145         scala.collection.Set<Object> measurements = IOUtils.invoked( Predef$.MODULE$.wrapRefArray( measurementFiles ) );
146         coverage.apply( measurements );
147 
148         int branchCount = coverage.branchCount();
149         int statementCount = coverage.statementCount();
150         int invokedBranchesCount = coverage.invokedBranchesCount();
151         int invokedStatementCount = coverage.invokedStatementCount();
152         getLog().debug( String.format( "invokedBranchesCount:%d / branchCount:%d, invokedStatementCount:%d / statementCount:%d",
153                                       invokedBranchesCount, branchCount, invokedStatementCount, statementCount ) );
154         if ( minimumCoverage > 0.0 )
155         {
156             if ( is100( minimumCoverage ) && is100( coverage.statementCoveragePercent() ) )
157             {
158                 getLog().info( "[scoverage] 100% Coverage !" );
159             }
160             else if ( minimumCoverage > coverage.statementCoveragePercent() )
161             {
162                 getLog().error( String.format( "[scoverage] Coverage is below minimum [%s%% < %.2f%%]",
163                                                coverage.statementCoverageFormatted(), minimumCoverage ) );
164                 if ( failOnMinimumCoverage )
165                 {
166                     throw new MojoFailureException( "Coverage minimum was not reached" );
167                 }
168             }
169             else
170             {
171                 getLog().info( String.format( "[scoverage] Coverage is above minimum [%s%% > %.2f%%]",
172                                               coverage.statementCoverageFormatted(), minimumCoverage ) );
173             }
174         }
175         getLog().info( String.format( "[scoverage] All done. Coverage was [%s%%]",
176                                       coverage.statementCoverageFormatted() ) );
177 
178         long te = System.currentTimeMillis();
179         getLog().debug( String.format( "Mojo execution time: %d ms", te - ts ) );
180     }
181 
182     // Private utility methods
183 
184     private boolean is100( Double d )
185     {
186         return Math.abs( 100 - d ) <= 0.00001d;
187     }
188
