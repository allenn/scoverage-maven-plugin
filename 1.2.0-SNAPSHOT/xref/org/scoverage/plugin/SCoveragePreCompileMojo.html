43  /**
44   * Configures project for compilation with SCoverage instrumentation.
45   * <br>
46   * <br>
47   * Supported compiler plugins:
48   * <ul>
49   * <li><a href="https://davidb.github.io/scala-maven-plugin/">net.alchim31.maven:scala-maven-plugin</a></li>
50   * <li><a href="https://code.google.com/p/sbt-compiler-maven-plugin/">com.google.code.sbt-compiler-maven-plugin:sbt-compiler-maven-plugin</a></li>
51   * </ul>
52   * <br>
53   * This is internal mojo, executed in forked {@code cobertura} life cycle.
54   * <br>
55   * 
56   * @author <a href="mailto:gslowikowski@gmail.com">Grzegorz Slowikowski</a>
57   * @since 1.0.0

180     /**
181      * Configures project for compilation with SCoverage instrumentation.
182      * 
183      * @throws MojoExecutionException if unexpected problem occurs
184      */
185     @Override
186     public void execute() throws MojoExecutionException
187     {
188         if ( "pom".equals( project.getPackaging() ) )
189         {
190             getLog().info( "Skipping SCoverage execution for project with packaging type 'pom'" );
191             //for aggragetor mojo - list of submodules: List<MavenProject> modules = project.getCollectedProjects();
192             return;
193         }
194 
195         if ( skip )
196         {
197             getLog().info( "Skipping Scoverage execution" );
198 
199             Properties projectProperties = project.getProperties();
200 
201             // for maven-resources-plugin (testResources), maven-compiler-plugin (testCompile),
202             // sbt-compiler-maven-plugin (testCompile), scala-maven-plugin (testCompile),
203             // maven-surefire-plugin and scalatest-maven-plugin
204             setProperty( projectProperties, "maven.test.skip", "true" );
205             // for scalatest-maven-plugin and specs2-maven-plugin
206             setProperty( projectProperties, "skipTests", "true" );
207 
208             return;
209         }
210 
211         long ts = System.currentTimeMillis();
212 
213         String scalaMainVersion = null;
214         String resolvedScalaVersion = resolveScalaVersion();
215         if ( resolvedScalaVersion != null )
216         {
217             if ( resolvedScalaVersion.startsWith( "2.10." ) )
218             {
219                 scalaMainVersion = "2.10";
220             }
221             else if ( resolvedScalaVersion.startsWith( "2.11." ) )
222             {
223                 scalaMainVersion = "2.11";
224             }
225             else
226             {
227                 getLog().warn( String.format( "Skipping SCoverage execution - unsupported Scala version \"%s\"",
228                                               resolvedScalaVersion ) );
229                 return;
230             }
231         }
232         else
233         {
234             getLog().warn( "Skipping SCoverage execution - Scala version not set" );
235             return;
236         }
237 
238         SCoverageForkedLifecycleConfigurator.afterForkedLifecycleEnter( project, reactorProjects );
239 
240         try
241         {
242             Artifact pluginArtifact = getScalaScoveragePluginArtifact( scalaMainVersion );
243             Artifact runtimeArtifact = getScalaScoverageRuntimeArtifact( scalaMainVersion );
244 
245             if ( pluginArtifact == null )
246             {
247                 return; // scoverage plugin will not be configured
248             }
249 
250             addScoverageDependenciesToTestClasspath( runtimeArtifact );
251 
252             String arg = DATA_DIR_OPTION + dataDirectory.getAbsolutePath();
253             String _scalacOptions = quoteArgument( arg );
254             String addScalacArgs = arg;
255 
256             if ( !StringUtils.isEmpty( excludedPackages ) )
257             {
258                 arg = EXCLUDED_PACKAGES_OPTION + excludedPackages.replace( "(empty)", "<empty>" );
259                 _scalacOptions = _scalacOptions + SPACE + quoteArgument( arg );
260                 addScalacArgs = addScalacArgs + PIPE + arg;
261             }
262 
263             if ( !StringUtils.isEmpty( excludedFiles ) )
264             {
265                 arg = EXCLUDED_FILES_OPTION + excludedFiles;
266                 _scalacOptions = _scalacOptions + SPACE + quoteArgument( arg );
267                 addScalacArgs = addScalacArgs + PIPE + arg;
268             }
269 
270             if ( highlighting )
271             {
272                 _scalacOptions = _scalacOptions + SPACE + "-Yrangepos";
273                 addScalacArgs = addScalacArgs + PIPE + "-Yrangepos";
274             }
275 
276             String _scalacPlugins =
277                 String.format( "%s:%s:%s", pluginArtifact.getGroupId(), pluginArtifact.getArtifactId(),
278                                pluginArtifact.getVersion() );
279 
280             arg = PLUGIN_OPTION + pluginArtifact.getFile().getAbsolutePath();
281             addScalacArgs = addScalacArgs + PIPE + arg;
282 
283             Properties projectProperties = project.getProperties();
284 
285             // for sbt-compiler-maven-plugin (version 1.0.0-beta5+)
286             setProperty( projectProperties, "sbt._scalacOptions", _scalacOptions );
287             // for sbt-compiler-maven-plugin (version 1.0.0-beta5+)
288             setProperty( projectProperties, "sbt._scalacPlugins", _scalacPlugins );
289             // for scala-maven-plugin (version 3.0.0+)
290             setProperty( projectProperties, "addScalacArgs", addScalacArgs );
291             // for scala-maven-plugin (version 3.1.0+)
292             setProperty( projectProperties, "analysisCacheFile",
293                          "${project.build.directory}/scoverage-analysis/compile" );
294             // for maven-surefire-plugin and scalatest-maven-plugin
295             setProperty( projectProperties, "maven.test.failure.ignore", "true" );
296 
297             // for maven-jar-plugin
298             // VERY IMPORTANT! Prevents from overwriting regular project artifact file
299             // with instrumented one during "integration-check" or "integration-report" execution.
300             project.getBuild().setFinalName( "scoverage-" + project.getBuild().getFinalName() );
301         }
302         catch ( ArtifactNotFoundException e )
303         {
304             throw new MojoExecutionException( "SCoverage preparation failed", e );
305         }
306         catch ( ArtifactResolutionException e )
307         {
308             throw new MojoExecutionException( "SCoverage preparation failed", e );
309         }
310 
311         long te = System.currentTimeMillis();
312         getLog().debug( String.format( "Mojo execution time: %d ms", te - ts ) );
313     }

317     private static final String SCALA_LIBRARY_GROUP_ID = "org.scala-lang";
318     private static final String SCALA_LIBRARY_ARTIFACT_ID = "scala-library";
319 
320     private static final String DATA_DIR_OPTION = "-P:scoverage:dataDir:";
321     private static final String EXCLUDED_PACKAGES_OPTION = "-P:scoverage:excludedPackages:";
322     private static final String EXCLUDED_FILES_OPTION = "-P:scoverage:excludedFiles:";
323     private static final String PLUGIN_OPTION = "-Xplugin:";
324 
325     private static final char DOUBLE_QUOTE = '\"';
326     private static final char SPACE = ' ';
327     private static final char PIPE = '|';
328 
329     private String quoteArgument( String arg )
330     {
331         return arg.indexOf( SPACE ) >= 0 ? DOUBLE_QUOTE + arg + DOUBLE_QUOTE : arg;
332     }
333 
334     private String resolveScalaVersion()
335     {
336         String result = scalaVersion;
337         if ( result == null )
338         {
339             // check project direct dependencies (transitive dependencies cannot be checked in this Maven lifecycle phase)
340             @SuppressWarnings("unchecked")
341             List<Dependency> dependencies = project.getDependencies();
342             for ( Dependency dependency: dependencies )
343             {
344                 if ( SCALA_LIBRARY_GROUP_ID.equals( dependency.getGroupId() )
345                     && SCALA_LIBRARY_ARTIFACT_ID.equals( dependency.getArtifactId() ) )
346                 {
347                     result = dependency.getVersion();
348                     break;
349                 }
350             }
351         }
352         return result;
353     }
354 
355     private void setProperty( Properties projectProperties, String propertyName, String newValue )
356     {
357         if ( projectProperties.containsKey( propertyName ) )
358         {
359             String oldValue = projectProperties.getProperty( propertyName );
360             projectProperties.put( "scoverage.backup." + propertyName, oldValue );
361         }
362         else
363         {
364             projectProperties.remove( "scoverage.backup." + propertyName );
365         }
366 
367         if ( newValue != null )
368         {
369             projectProperties.put( propertyName, newValue );
370         }
371         else
372         {
373             projectProperties.remove( propertyName );
374         }
375     }

377     private Artifact getScalaScoveragePluginArtifact( String scalaMainVersion )
378         throws ArtifactNotFoundException, ArtifactResolutionException
379     {
380         Artifact result = null;
381 
382         String resolvedScalacPluginVersion = scalacPluginVersion;
383         if ( resolvedScalacPluginVersion == null || "".equals( resolvedScalacPluginVersion ) )
384         {
385             for ( Artifact artifact : pluginArtifacts )
386             {
387                 if ( "org.scoverage".equals( artifact.getGroupId() )
388                     && "scalac-scoverage-plugin_2.10".equals( artifact.getArtifactId() ) )
389                 {
390                     if ( "2.10".equals( scalaMainVersion ) )
391                     {
392                         return artifact; // shortcut, use the same artifact plugin uses
393                     }
394                     resolvedScalacPluginVersion = artifact.getVersion();
395                     break;
396                 }
397             }
398         }
399 
400         result =
401             getResolvedArtifact( "org.scoverage", "scalac-scoverage-plugin_" + scalaMainVersion,
402                                  resolvedScalacPluginVersion );
403         return result;
404     }

406     private Artifact getScalaScoverageRuntimeArtifact( String scalaMainVersion )
407         throws ArtifactNotFoundException, ArtifactResolutionException
408     {
409         Artifact result = null;
410 
411         String resolvedScalacRuntimeVersion = scalacPluginVersion;
412         if ( resolvedScalacRuntimeVersion == null || "".equals( resolvedScalacRuntimeVersion ) )
413         {
414             for ( Artifact artifact : pluginArtifacts )
415             {
416                 if ( "org.scoverage".equals( artifact.getGroupId() )
417                     && "scalac-scoverage-plugin_2.10".equals( artifact.getArtifactId() ) )
418                 {
419                     resolvedScalacRuntimeVersion = artifact.getVersion();
420                     break;
421                 }
422             }
423         }
424 
425         result =
426             getResolvedArtifact( "org.scoverage", "scalac-scoverage-runtime_" + scalaMainVersion,
427                                  resolvedScalacRuntimeVersion );
428         return result;
429     }

450     /**
451      * Use provided instead of just test, so it's available on both compile and test classpath (MCOBERTURA-26)
452      *
453      * @param artifact
454      * @return re-scoped artifact
455      */
456     private Artifact artifactScopeToProvided( Artifact artifact )
457     {
458         return factory.createArtifact( artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion(),
459                                        Artifact.SCOPE_PROVIDED, artifact.getType() );
460     }

463     private Artifact getResolvedArtifact( String groupId, String artifactId, String version )
464         throws ArtifactNotFoundException, ArtifactResolutionException
465     {
466         Artifact artifact = factory.createArtifact( groupId, artifactId, version, Artifact.SCOPE_RUNTIME, "jar" );
467         resolver.resolve( artifact, remoteRepos, localRepo );
468         return artifact;
469     }
